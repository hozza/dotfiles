# vim Hints

F2		=NERDTree 
F9		=TagBar
F7|F8	=spellcheck, fix

za		=toggle code fold
g;		=jump to last edit

jj		=esc/normal mode
o/O		=inset new line under/over
A		=inset end of line

Tab		=next buff
S-Tab	=prev buff
\dd 	=del buff

\b		=fuz buffers
\f|\g	=fuz files, git files
\l|\t	=fuz lines, tags

\x|\X	=replace, back/forward

u		=undo 
C+r		=redo 
#dd		=del 
y		=copy
#yy		=copy line
p		=paste 
J		=join 

~		=case swap
*|\		=search/find
%s		= replace
zg		=add spell to dic 

v		=visual mode
><		=indent

C+o		=single command 
i		=insert 

:x		=save
:saveas =make copy
:b#		=buffer number
:bn		=buffer next
:bp		=buffer prev

:sp		=split view
:vsp	=vertical split

:so $MYVIMRC	= reload settings in-place
Ctrl+o			= run normal mode command once

]s 		= next spelling error
[s 		= previous spelling error
z= 		= show suggestions
zg 		= adding spelling to dictionary
zug 	= remove word from dictionary

q[a-z]	= macros
"[a-z]	= registers

Cw+=	= reset Goyo size
